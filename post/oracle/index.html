#dbms_lob.substr()函数：将大文本转换字符类型读出来
SELECT dbms_lob.substr(EDUCATION) EDUCATION,dbms_lob.substr(INTERSHIP) INTERSHIP,dbms_lob.substr(EVALUATION) EVALUATION FROM T_EXAM_RESUME_YANGCHENG


#判定长度   DBMS_LOB.GETLENGTH(CLOB字段)
#获取文本   DBMS_LOB.SUBSTR(CLOB字段,n,pos)
DBMS_LOB.SUBSTR(EDUCATION,1,1)表示从EDUCATION的第1个字节开始取出1个字节
DBMS_LOB.SUBSTR(CLOB)表示截取CLOB变量保存的全部数据

例：EVALUATION 内容：还可以
SELECT DBMS_LOB.GETLENGTH(EVALUATION) EVALUATION FROM T_EXAM_RESUME_YANGCHENG -->结果：3

例：EDUCATION 内容：暂无
SELECT DBMS_LOB.SUBSTR(EDUCATION,1,1) EDUCATION FROM T_EXAM_RESUME_YANGCHENG  -->结果：暂

在使用Union all连接时，若A集合中某列为nvarchar2或nvarchar类型，而B集合中无此列，用 ’来代替是会报字符集不匹配。

#LISTAGG函数
#LISTAGG(XXX,'$') WITHIN GROUP(ORDER BY XXX) 

SELECT PKID, DATA_TYPES_CODE ,LISTAGG(ZYLYBM_R,'$') WITHIN GROUP(ORDER BY ZYLYBM_R) ,LISTAGG(ZYLYMC_R,'$') WITHIN GROUP(ORDER BY ZYLYMC_R) 
FROM HISTORY.AAA_YANGCHENG_ZDLY GROUP BY ( PKID, DATA_TYPES_CODE)

# 开发大屏库
SELECT KEY,LISTAGG(VALUE,'$') WITHIN GROUP(ORDER BY VALUE) VALUE FROM (
SELECT STANDENTERID KEY,CASE WHEN ENTER_TYPE ='jp' THEN '减排企业' ELSE '白名单企业' END AS  VALUE  FROM THS_DONGAOBZ.T_BAS_GKQD_ENTERPRISE
WHERE STANDENTERID IS NOT NULL GROUP BY STANDENTERID,ENTER_TYPE
UNION ALL 
SELECT STANDENTERID KEY ,CASE WHEN IS_SSGK = '1' THEN '赛事管控企业' ELSE '' END AS  VALUE FROM THS_DONGAOBZ.T_BAS_GKQD_ENTERPRISE
WHERE STANDENTERID IS NOT NULL GROUP BY STANDENTERID,IS_SSGK
) GROUP BY KEY
