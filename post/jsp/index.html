JSP 与 PHP、ASP、ASP.NET 等语言类似，运行在服务端的语言。

JSP（全称Java Server Pages）是由 Sun Microsystems 公司倡导和许多公司参与共同创建的一种使软件开发者可以响应客户端请求，而动态生成 HTML、XML 或其他格式文档的Web网页的技术标准。

JSP 技术是以 Java 语言作为脚本语言的，JSP 网页为整个服务器端的 Java 库单元提供了一个接口来服务于HTTP的应用程序。

JSP文件后缀名为 *.jsp 。

JSP开发的WEB应用可以跨平台使用，既可以运行在 Linux 上也能运行在 Windows 上。

@RequestMapping("/province")
    public ModelAndView province(Paging<Map<String, Object>> pageInfo,
            FormModel formModel) {
         ModelAndView model = new ModelAndView("countsummary/taskcompletion/province");
         Map<String, Object> data = formModel.getForm();
         Paging<Map<String, Object>> mapPaging  = taskCompletionService.listPageProvince(pageInfo, data);
         //获取任务完成情况——各省情况轮次信息
         model.addObject("alternately", taskCompletionService.alternatelyProvinceList(data));
         //获取任务完成情况——各省情况调度月份信息
         model.addObject("dispatchmonth", taskCompletionService.dispatchmonthProvinceList(data));
        //获取任务完成情况——各省情况数据来源信息  （这个数据来源！！！）
         model.addObject("datasource", taskCompletionService.datasourceProvinceList(data));
         model.addObject("pageInfo",mapPaging);
         model.addObject("form", formModel.getForm());
        return model;
    }

JS代码:
<script>
$(function () {
        topChange();
    })

    function topCheck(checked) {
        var che = $("tbody").find("input[type='checkbox']");
        for(var key = 0; key < che.length; key++){
            che[key].checked = checked;
            dwCheck(che[key].value, checked);
        }
    }

    function dwCheck(value, check) {
        var codenames = value.split(":");
        for (let i = 0; i < codenames.length; i++) {
            var code = codenames[0];
            var name = codenames[1];
        }
        var values = $("#TRANSFERS").val();
        var dws = [];
        if (values != null && values != '') {
            dws = values.split(",");
        }
        if (check) {
            //去重判断
            if(values.indexOf(value) == -1){
                dws.push(value);
                //调用添加到UL方法
                addSelectToUL(code, name);
            }
        } else {
            dws.remove(value);
            removeSelectFromUL(code);
        }
        $("#TRANSFERS").val(dws.join(","));
        topChange();
    }

    function topChange() {
        var che = $("tbody").find("input[type='checkbox']");
        var top = $("thead").find("input[type='checkbox']");
        var checked = true;
        for (var key in che) {
            if (che[key].checked == false) {
                checked = false;
                break;
            }
        }
        top[0].checked = checked;
    }
</script>

// 字段类型名称切换获取选项类型
    function doFieldSwitch(){
        // 先清空选项类型下拉框的值
        $("#id_dict_code").empty();
        ths.submitFormAjax({
            url:'doswitch.vm',
            data: {"form[FIELD_TYPE_CODE]": $("#id_filed_type_code").val()}
            ,success:function (resp) {
                var data = eval(resp);
                var html = "<option value=''>--请选择--</option>";
                for(var i=0; i<data.length; i++){
                    html += "<option value='"+ data[i].DICT_CODE+"'>"+ data[i].DICT_NAME+"</option>";
                }
                // 给选项类型下拉框添加值
                $("#id_dict_code").append(html);
            }
        });
    }
