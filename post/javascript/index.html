JS显示Uncaught TypeError: XXX is not a function

1
2
3

原因：button控件的id名和JS的方法名一样，执行的时候先找到的是HTML控件
 例：<button type="button" class="btn btn-xs btn-xs-ths"  onclick="doImport()" id="doImport"> </button>
解决：修改id名称不与JS方法同名即可。

// JSON.parse()将JSON字符串转为一个对象,parse用于从一个字符串中解析出json对象
var str = '{"name":"huangxiaojian","age":"23"}' 
console.log(JSON.parse(str))//Object {name: "huangxiaojian", age: "23"}

// JSON.stringify()是将 JavaScript 对象转换为 JSON 字符串,stringify()用于从一个对象解析出字符串
var a = {a:1,b:2}
console.log(JSON.stringify(a))//{"a":1,"b":2}
